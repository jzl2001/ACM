const int N = 1e6 + 10;
bool vis[N];
int main(){
	string s;
	cin >> s;
	int len = s.size();
	int l = 0, r = len - 1;
	for (;;)
	{
		if (r - l + 1 < 4)
		{
			vis[r] = 1;
			break;
		}
		if (s[l] == s[r])
			vis[l] = 1, vis[r] = 1;
		else
			if (s[l] == s[r - 1])
				vis[l] = 1, vis[r - 1] = 1;
			else
				if (s[l + 1] == s[r])
					vis[l + 1] = 1, vis[r] = 1;
				else
					vis[l + 1] = 1, vis[r - 1] = 1;
		l += 2, r -= 2;
	}
	for (int i = 0; i < len; i++)
	{
		if (vis[i] == 1)
			cout << s[i];
	}
	return 0;
}
