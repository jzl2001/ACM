#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 10;
typedef long long ll;
struct node{
	int x, y;
	bool operator < (const node &oth)const
	{
		if (x != oth.x)
			return x < oth.x;
		else
			return y > oth.y;
	}
}p[N];
int main(){
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		for (int i = 1; i <= n; i++)
			scanf("%d%d", &p[i].x, &p[i].y);
		sort(p + 1, p + n + 1);
		ll a, b, c, d;
		a = 1, b = 9e8 - 1, c = -1, d = -9e8;
		cout << a + p[n / 2].x << " " << b + p[n / 2].y << " " << c + p[n / 2].x << " " << d + p[n / 2].y << endl;
	}
}
